## Install

```bash
npm install --save sample-library
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'sample-library'
import 'sample-library/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
